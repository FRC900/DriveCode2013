008package edu.wpi.first.wpilibj.templates;
009
010import edu.wpi.first.wpilibj.CANJaguar;
011import edu.wpi.first.wpilibj.IterativeRobot;
012import edu.wpi.first.wpilibj.Joystick;
013
014/**
015 * The VM is configured to automatically run this class, and to call the
016 * functions corresponding to each mode, as described in the IterativeRobot
017 * documentation. If you change the name of this class or the package after
018 * creating this project, you must also update the manifest file in the resource
019 * directory.
020 * 
021 * @author Andrew Vitkus
022 */
023public class FinalRobot extends IterativeRobot {
024    Joystick controller1;
025    
026    CANDrive drive;
027    
028    Shooter shooter;
029    
030    Lift lift;
031    
032    double teleopIterCount;
033    double autoIterCount;
034    
035    /**
036     * This function is run when the robot is first started up and should be
037     * used for any initialization code.
038     */
039    public void robotInit() {
040        controller1 = new Joystick(1);  // first joystick setup as controller 1
041        drive = new CANDrive(1,2,3,4);  // setup drive train on CAN adresses 1, 2, 3, and 4
042        lift = new Lift(1, 2);           // setup frisbee lift with front polycord motor on PWM 1 and rear on PWM 2
043        shooter = new Shooter(5, 4, 3);    // setup shooter with wheel on CAN address 5, lift on PWM 3
044        
045        drive.setDeadBand(.1);
046    }
047    
048    /**
049     * This function is run when the robot first enters the autonomous period
050     */
051    public void autonomousInit() {
052        drive.setControlMode(CANJaguar.ControlMode.kPosition);  // set the drive train to run based on position data
053        autoIterCount = 0;  // zero the autonomous iteration counter
054    }
055
056    /**
057     * This function is called periodically during autonomous
058     */
059    public void autonomousPeriodic() {
060
061    }
062    
063    /**
064     * This function is called when the robot first enters operator control
065     */
066    public void teleopInit() {
067        drive.setControlMode(CANJaguar.ControlMode.kPercentVbus );  // set the drive train to run based on %Vbus
068        teleopIterCount = 0;    // zero the telop iteration counter
069    }
070
071    /**
072     * This function is called periodically during operator control
073     */
074    public void teleopPeriodic() {
075        drive.tankDrive(controller1.getRawAxis(2) * -1, controller1.getRawAxis(4) * -1, true);    // drive with left Y-axis driving the left motors and right Y-axis driving the right
076
077        if (controller1.getRawAxis(6) == 1) {
078            lift.lift();    // if the D-pad is up, run the lift up
079        } else if (controller1.getRawAxis(6) == -1) {
080            lift.reverse(); // if the D-pad is down, run the lift backwards
081        } else if (controller1.getRawAxis(5) == 1) {
082            lift.stop();    //if the D-pad is right, stop the lift
083        }
084        
085        if (teleopIterCount % 5 == 0) {
086            if (controller1.getRawButton(5)) { // angle the shooter up if the left bumper is pressed
087                shooter.raise();
088            } else if (controller1.getRawButton(6)) {   // angle the shooter down if the right bumper is pressed
089                shooter.lower();
090            } else {    // if neither bumper is pressed stop angling the shooter
091                shooter.stopLift();
092            }
093            
094            if (controller1.getRawButton(8)) { // if right trigger is pressed, increase the shooter speed by 5% V-bus
095                shooter.raiseShooterSpeed(.05); 
096            } else if (controller1.getRawButton(7)) { // if left trigger is pressed, decrease the shooter speed by 5% V-bus
097                shooter.lowerShooterSpeed(.05); 
098            } else if (controller1.getRawButton(2)) { // if the B button is pressed, stop the shooter
099                shooter.stopShooter();  
100            }
101        }
102        
103        if (controller1.getRawButton(3)) {
104            shooter.shoot();    // if A button is pressed, shoot
105        }
106        
107        teleopIterCount ++; // increments the teleop iteration counter
108    }
109    
110    /**
111     * This function is called periodically during test mode
112     */
113    public void testPeriodic() {
114    
115    }
116    
117}
